function handleSalaryInput() { const el = document.getElementById('salaryInput'), val = el.value.trim().toLowerCase(); if (!val) return; if (val === 'a') { showAllTotal(); el.value = ''; el.blur(); updateClearBtn(); } else if (val.startsWith('g')) { if(!showGraph(val)) showInvalidModal(); el.value = ''; el.blur(); updateClearBtn(); } else { const m = el.value.trim().match(/^(\d+)\/(\d+)\/(\d+(?:\+\d+)*)(?:\/(.*))?$/); if (!m) { showInvalidModal(); el.value = ''; el.blur(); updateClearBtn(); return; } const y = parseInt(m[1]), mo = parseInt(m[2]); if (mo < 1 || mo > 12 || y < startYear || y > endYear) { showInvalidModal(); el.value = ''; el.blur(); updateClearBtn(); return; } const key = `${y}/${mo}`, rawInput = el.value.trim(); if (salaryData[key] && !isSalaryEditMode) { openConfirmModal("Overwrite?", "Existing record found for this month.", "Overwrite", "blue", () => { if(saveSalaryRecord(rawInput)) showConfetti(); }); el.value = ''; el.blur(); updateClearBtn(); } else { if(!saveSalaryRecord(rawInput)) showInvalidModal(); else showConfetti(); el.value = ''; el.blur(); updateClearBtn(); } } } function saveSalaryRecord(input) { const m = input.match(/^(\d+)\/(\d+)\/(\d+(?:\+\d+)*)(?:\/(.*))?$/); if (!m) return false; const y = parseInt(m[1]), mo = parseInt(m[2]); const amt = m[3].split('+').reduce((sum, val) => sum + parseInt(val), 0); const memo = m[4] || ''; if (mo < 1 || mo > 12 || y < startYear || y > endYear) return false; lastSavedSalaryKey = `${y}/${mo}`; salaryData[lastSavedSalaryKey] = { amount: amt, memo: memo }; localStorage.setItem('myTenYearSalaryData', JSON.stringify(salaryData)); initSalaryCalendar(y); isSalaryEditMode = false; return true; }
